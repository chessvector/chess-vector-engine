34#[allow(dead_code)]
35impl GameState {
36    fn new(chess_vector_color: Color) -> Self {
37        Self {
38            game: Game::new(),
39            move_history: Vec::new(),
40            chess_vector_color,
41            start_time: Utc::now(),
42        }
43    }
44
45    fn current_board(&self) -> Board {
46        self.game.current_position()
47    }
48
49    fn make_move(&mut self, analysis: MoveAnalysis) -> Result<(), &'static str> {
50        if self.game.make_move(analysis.chess_move) {
51            self.move_history.push(analysis);
52            Ok(())
53        } else {
54            Err("Invalid move")
55        }
56    }
57
58    fn is_chess_vector_turn(&self) -> bool {
59        self.current_board().side_to_move() == self.chess_vector_color
60    }
61
62    fn calculate_statistics(&self) -> (f32, f32, f32, f32) {
63        let chess_vector_moves: Vec<_> = self
64            .move_history
65            .iter()
66            .enumerate()
67            .filter(|(i, _)| {
68                let move_color = if i % 2 == 0 {
69                    Color::White
70                } else {
71                    Color::Black
72                };
73                move_color == self.chess_vector_color
74            })
75            .map(|(_, analysis)| analysis)
76            .collect();
77
78        let stockfish_moves: Vec<_> = self
79            .move_history
80            .iter()
81            .enumerate()
82            .filter(|(i, _)| {
83                let move_color = if i % 2 == 0 {
84                    Color::White
85                } else {
86                    Color::Black
87                };
88                move_color != self.chess_vector_color
89            })
90            .map(|(_, analysis)| analysis)
91            .collect();
92
93        let chess_vector_total_loss: f32 =
94            chess_vector_moves.iter().map(|m| m.centipawn_loss).sum();
95        let chess_vector_good_moves = chess_vector_moves
96            .iter()
97            .filter(|m| m.centipawn_loss < 20.0)
98            .count();
99        let chess_vector_accuracy = if chess_vector_moves.is_empty() {
100            0.0
101        } else {
102            chess_vector_good_moves as f32 / chess_vector_moves.len() as f32 * 100.0
103        };
104
105        let stockfish_total_loss: f32 = stockfish_moves.iter().map(|m| m.centipawn_loss).sum();
106        let stockfish_good_moves = stockfish_moves
107            .iter()
108            .filter(|m| m.centipawn_loss < 20.0)
109            .count();
110        let stockfish_accuracy = if stockfish_moves.is_empty() {
111            0.0
112        } else {
113            stockfish_good_moves as f32 / stockfish_moves.len() as f32 * 100.0
114        };
115
116        (
117            chess_vector_total_loss,
118            chess_vector_accuracy,
119            stockfish_total_loss,
120            stockfish_accuracy,
121        )
122    }
123}

449/// Analyze material balance
450fn analyze_material(board: &Board) {
451    println!("\n=== Material Analysis ===");
452
453    use chess::{Color, Piece};
454
455    let pieces = [
456        Piece::Pawn,
457        Piece::Knight,
458        Piece::Bishop,
459        Piece::Rook,
460        Piece::Queen,
461    ];
462    let values = [1, 3, 3, 5, 9];
463
464    let mut white_material = 0;
465    let mut black_material = 0;
466
467    for (piece, value) in pieces.iter().zip(values.iter()) {
468        let white_count = (board.pieces(*piece) & board.color_combined(Color::White)).popcnt();
469        let black_count = (board.pieces(*piece) & board.color_combined(Color::Black)).popcnt();
470
471        white_material += white_count * value;
472        black_material += black_count * value;
473
474        if white_count != black_count {
475            println!(
476                "{:?}: White {} vs Black {} (difference: {:+})",
477                piece,
478                white_count,
479                black_count,
480                white_count as i32 - black_count as i32
481            );
482        }
483    }
484
485    let material_balance = white_material as i32 - black_material as i32;
486    println!(
487        "Total material: White {} vs Black {} (balance: {:+})",
488        white_material, black_material, material_balance
489    );
490}

496    use chess::Color;

502        let rank = king_sq.get_rank().to_index();
503
504        let safety_desc = match (color, file, rank) {
505            (Color::White, 0..=2, 0..=1) => "Queenside castle",
506            (Color::White, 5..=7, 0..=1) => "Kingside castle",
507            (Color::Black, 0..=2, 6..=7) => "Queenside castle",
508            (Color::Black, 5..=7, 6..=7) => "Kingside castle",
509            (_, 3..=4, _) => "Exposed in center",
510            _ => "Custom position",
511        };
512
513        println!("{:?} king: {} ({})", color, king_sq, safety_desc);
514    }
515
516    // Development
517    let white_developed = count_developed_pieces(board, Color::White);
518    let black_developed = count_developed_pieces(board, Color::Black);

521        "Developed pieces: White {} vs Black {}",
522        white_developed, black_developed

526/// Count developed pieces (not on starting squares)
527fn count_developed_pieces(board: &Board, color: Color) -> u32 {
528    use chess::Square;
529
530    let mut developed = 0;
531
532    // Check if minor pieces have moved from starting squares
533    let back_rank = if color == Color::White { 0 } else { 7 };
534    let starting_squares = [
535        Square::make_square(chess::Rank::from_index(back_rank), chess::File::B),
536        Square::make_square(chess::Rank::from_index(back_rank), chess::File::C),
537        Square::make_square(chess::Rank::from_index(back_rank), chess::File::F),
538        Square::make_square(chess::Rank::from_index(back_rank), chess::File::G),
539    ];
540
541    for square in starting_squares {
542        let sq = square;
543        if board.piece_on(sq).is_none() || board.color_on(sq) != Some(color) {
544            developed += 1;
545        }
546    }
547
548    developed
