240mod license_utils {
241    use std::time::{SystemTime, UNIX_EPOCH};
242
243    #[allow(dead_code)]
244    pub fn current_timestamp() -> u64 {
245        SystemTime::now()
246            .duration_since(UNIX_EPOCH)
247            .unwrap_or_default()
248            .as_secs()
249    }
