91/// Generate a semi-random chess position by playing random moves
92fn generate_random_position(rng: &mut impl Rng) -> Result<Board, Box<dyn std::error::Error>> {
93    let mut board = Board::default();
94    let moves_to_play = rng.gen_range(0..20);
95
96    for _ in 0..moves_to_play {
97        let legal_moves: Vec<_> = MoveGen::new_legal(&board).collect();
98        if legal_moves.is_empty() {
99            break;
100        }
101
102        let random_move = legal_moves[rng.gen_range(0..legal_moves.len())];
103        board = board.make_move_new(random_move);
104    }
105
106    Ok(board)
107}

109/// Test similarity calculation accuracy with known similar/dissimilar positions
110fn test_similarity_accuracy(engine: &ChessVectorEngine) {
111    use std::str::FromStr;
112
113    // Test cases: positions that should be similar vs dissimilar
114    let similar_pairs = vec![
115        // Starting position vs 1.e4
116        (
117            "rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1",
118            "rnbqkbnr/pppppppp/8/8/4P3/8/PPPP1PPP/RNBQKBNR b KQkq e3 0 1",
119        ),
120        // 1.e4 e5 vs 1.d4 d5 (symmetric openings)
121        (
122            "rnbqkbnr/pppp1ppp/8/4p3/4P3/8/PPPP1PPP/RNBQKBNR w KQkq e6 0 2",
123            "rnbqkbnr/ppp1pppp/8/3p4/3P4/8/PPP1PPPP/RNBQKBNR w KQkq d6 0 2",
124        ),
125    ];
126
127    let dissimilar_pairs = vec![
128        // Opening vs endgame
129        (
130            "rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1",
131            "4k3/8/8/8/8/8/4P3/4K3 w - - 0 1",
132        ),
133        // Opening vs middlegame with major piece differences
134        (
135            "rnbqkbnr/pppppppp/8/8/4P3/8/PPPP1PPP/RNBQKBNR b KQkq e3 0 1",
136            "r1bq1rk1/pppp1ppp/2n2n2/2b1p3/2B1P3/3P1N2/PPP2PPP/RNBQK2R w KQ - 4 6",
137        ),
138    ];
139
140    println!("Testing similarity on known similar pairs:");
141    for (pos1, pos2) in similar_pairs {
142        let board1 = Board::from_str(pos1).expect("Valid FEN");
143        let board2 = Board::from_str(pos2).expect("Valid FEN");
144        let similarity = engine.calculate_similarity(&board1, &board2);
145        println!("  Similarity: {:.3} (should be high)", similarity);
146    }
147
148    println!("Testing similarity on known dissimilar pairs:");
149    for (pos1, pos2) in dissimilar_pairs {
150        let board1 = Board::from_str(pos1).expect("Valid FEN");
151        let board2 = Board::from_str(pos2).expect("Valid FEN");
152        let similarity = engine.calculate_similarity(&board1, &board2);
153        println!("  Similarity: {:.3} (should be low)", similarity);
154    }
155}
