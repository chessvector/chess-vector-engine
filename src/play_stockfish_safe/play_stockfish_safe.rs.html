11/// Normalize evaluation to be from White's perspective and in reasonable units
12fn normalize_evaluation(raw_eval: f32, _side_to_move: Color) -> f32 {
13    // Convert from centipawns to pawns (tactical search returns centipawns)
14    let eval = raw_eval / 100.0;
15
16    // The tactical search now always returns evaluation from White's perspective
17    // (positive = good for White, negative = good for Black)
18    // No perspective conversion needed
19
20    // Clamp to reasonable range (-10 to +10 pawns)
21    eval.clamp(-10.0, 10.0)
22}

24/// Convert a ChessMove to Standard Algebraic Notation (SAN)
25fn convert_to_san(board: &Board, chess_move: ChessMove) -> String {
26    let from_square = chess_move.get_source();
27    let to_square = chess_move.get_dest();
28    let piece = board.piece_on(from_square);
29
30    if piece.is_none() {
31        return chess_move.to_string(); // Fallback to long algebraic notation
32    }
33
34    let piece = piece.unwrap();
35    let promotion = chess_move.get_promotion();
36
37    // Handle castling
38    if piece == Piece::King {
39        if from_square == Square::E1 && to_square == Square::G1 {
40            return "O-O".to_string();
41        }
42        if from_square == Square::E1 && to_square == Square::C1 {
43            return "O-O-O".to_string();
44        }
45        if from_square == Square::E8 && to_square == Square::G8 {
46            return "O-O".to_string();
47        }
48        if from_square == Square::E8 && to_square == Square::C8 {
49            return "O-O-O".to_string();
50        }
51    }
52
53    let mut san = String::new();
54
55    // Add piece letter (except for pawns)
56    if piece != Piece::Pawn {
57        san.push(match piece {
58            Piece::King => 'K',
59            Piece::Queen => 'Q',
60            Piece::Rook => 'R',
61            Piece::Bishop => 'B',
62            Piece::Knight => 'N',
63            Piece::Pawn => unreachable!(),
64        });
65    }
66
67    // Handle disambiguation for pieces (simplified)
68    if piece != Piece::Pawn {
69        let legal_moves: Vec<ChessMove> = MoveGen::new_legal(board).collect();
70        let same_piece_moves: Vec<ChessMove> = legal_moves
71            .iter()
72            .filter(|&mv| {
73                board.piece_on(mv.get_source()) == Some(piece)
74                    && mv.get_dest() == to_square
75                    && mv.get_source() != from_square
76            })
77            .cloned()
78            .collect();
79
80        if !same_piece_moves.is_empty() {
81            // Need disambiguation - use file if different files, otherwise use rank
82            let need_file_disambiguation = same_piece_moves
83                .iter()
84                .any(|mv| mv.get_source().get_file() != from_square.get_file());
85
86            if need_file_disambiguation {
87                san.push((b'a' + from_square.get_file().to_index() as u8) as char);
88            } else {
89                san.push((b'1' + from_square.get_rank().to_index() as u8) as char);
90            }
91        }
92    }
93
94    // Handle captures
95    let is_capture = board.piece_on(to_square).is_some()
96        || (piece == Piece::Pawn && from_square.get_file() != to_square.get_file());
97
98    if is_capture {
99        // For pawn captures, add the file of departure
100        if piece == Piece::Pawn {
101            san.push((b'a' + from_square.get_file().to_index() as u8) as char);
102        }
103        san.push('x');
104    }
105
106    // Add destination square
107    san.push((b'a' + to_square.get_file().to_index() as u8) as char);
108    san.push((b'1' + to_square.get_rank().to_index() as u8) as char);
109
110    // Handle promotion
111    if let Some(promo_piece) = promotion {
112        san.push('=');
113        san.push(match promo_piece {
114            Piece::Queen => 'Q',
115            Piece::Rook => 'R',
116            Piece::Bishop => 'B',
117            Piece::Knight => 'N',
118            _ => 'Q', // Default to queen
119        });
120    }
121
122    // Check for check/checkmate (simplified - just add if next position is in check)
123    let next_board = board.make_move_new(chess_move);
124    if next_board.checkers().0 != 0 {
125        // Simple check detection - would need full game state to determine checkmate
126        san.push('+');
127    }
128
129    san
130}

132/// Enhanced Stockfish process for interactive gameplay
133struct StockfishPlayer {

136    stdout: BufReader<std::process::ChildStdout>,

144impl StockfishPlayer {
145    fn new(depth: u8, time_per_move: u32) -> Result<Self, Box<dyn std::error::Error>> {
146        let mut child = ProcessCommand::new("stockfish")
147            .stdin(Stdio::piped())
148            .stdout(Stdio::piped())
149            .stderr(Stdio::piped())
150            .spawn()
151            .map_err(|e| {
152                format!(
153                    "Failed to start Stockfish. Make sure it's installed and in PATH. Error: {}",
154                    e
155                )
156            })?;
157
158        let stdin = BufWriter::new(
159            child
160                .stdin
161                .take()
162                .ok_or("Failed to get stdin handle for Stockfish process")?,
163        );
164        let stdout = BufReader::new(
165            child
166                .stdout
167                .take()
168                .ok_or("Failed to get stdout handle for Stockfish process")?,
169        );
170
171        let mut player = Self {
172            child,
173            stdin,
174            stdout,
175            name: "Stockfish".to_string(),
176            depth,
177            time_per_move,
178        };
179
180        // Initialize UCI
181        player.send_command("uci")?;
182        player.wait_for_uciok()?;
183        player.send_command("setoption name Hash value 256")?;
184        player.send_command("setoption name Threads value 1")?;
185        player.send_command("isready")?;
186        player.wait_for_readyok()?;
187
188        println!("âœ… Stockfish engine ready (depth {})", depth);
189
190        Ok(player)
191    }

223    fn get_best_move(
224        &mut self,
225        board: &Board,
226    ) -> Result<(ChessMove, f32), Box<dyn std::error::Error>> {
227        let fen = board.to_string();
228
229        // Send position
230        self.send_command(&format!("position fen {}", fen))?;
231
232        // Start search with time control
233        self.send_command(&format!("go movetime {}", self.time_per_move))?;
234
235        let mut line = String::new();
236        let mut best_move = None;
237        let mut evaluation = 0.0;
238
239        loop {
240            line.clear();
241            self.stdout.read_line(&mut line)?;
242            let line = line.trim();
243
244            if line.starts_with("info") && line.contains("depth") && line.contains("pv") {
245                // Extract evaluation
246                if line.contains("score cp") {
247                    if let Some(cp_pos) = line.find("score cp ") {
248                        let cp_str = &line[cp_pos + 9..];
249                        if let Some(end) = cp_str.find(' ') {
250                            if let Ok(cp_value) = cp_str[..end].parse::<i32>() {
251                                evaluation = cp_value as f32 / 100.0;
252                            }
253                        }
254                    }
255                } else if line.contains("score mate") {
256                    if let Some(mate_pos) = line.find("score mate ") {
257                        let mate_str = &line[mate_pos + 11..];
258                        if let Some(end) = mate_str.find(' ') {
259                            if let Ok(mate_moves) = mate_str[..end].parse::<i32>() {
260                                evaluation = if mate_moves > 0 { 100.0 } else { -100.0 };
261                            }
262                        }
263                    }
264                }
265
266                // Extract principal variation
267                if let Some(pv_pos) = line.find(" pv ") {
268                    let _pv_line = line[pv_pos + 4..].to_string();
269                }
270            } else if line.starts_with("bestmove") {
271                let parts: Vec<&str> = line.split_whitespace().collect();
272                if parts.len() >= 2 && parts[1] != "(none)" {
273                    best_move = ChessMove::from_str(parts[1]).ok();
274                }
275                break;
276            }
277        }
278
279        match best_move {
280            Some(mv) => Ok((mv, evaluation)),
281            None => Err("Stockfish did not return a valid move".into()),
282        }
283    }
284}

343    fn generate_pgn(&self) -> String {
344        let mut pgn = String::new();
345
346        // PGN headers
347        pgn.push_str("[Event \"Chess Vector vs Stockfish (Safe Mode)\"]\n");
348        pgn.push_str("[Site \"CLI Match\"]\n");
349        pgn.push_str(&format!(
350            "[Date \"{}\"]\n",
351            self.start_time.format("%Y.%m.%d")
352        ));
353        pgn.push_str("[Round \"1\"]\n");
354
355        match self.chess_vector_color {
356            Color::White => {
357                pgn.push_str("[White \"Chess Vector Engine\"]\n");
358                pgn.push_str("[Black \"Stockfish\"]\n");
359            }
360            Color::Black => {
361                pgn.push_str("[White \"Stockfish\"]\n");
362                pgn.push_str("[Black \"Chess Vector Engine\"]\n");
363            }
364        }
365
366        // Game result
367        let result = match self.game.result() {
368            Some(chess::GameResult::WhiteCheckmates) => "1-0",
369            Some(chess::GameResult::BlackCheckmates) => "0-1",
370            Some(chess::GameResult::WhiteResigns) => "0-1",
371            Some(chess::GameResult::BlackResigns) => "1-0",
372            Some(chess::GameResult::Stalemate) => "1/2-1/2",
373            Some(chess::GameResult::DrawAccepted) => "1/2-1/2",
374            Some(chess::GameResult::DrawDeclared) => "1/2-1/2",
375            None => "*",
376        };
377        pgn.push_str(&format!("[Result \"{}\"]\n\n", result));
378
379        // Moves with analysis annotations - using proper SAN notation
380        let mut move_number = 1;
381        let mut current_board = Board::default(); // Start with initial position
382
383        for (i, analysis) in self.move_history.iter().enumerate() {
384            if i % 2 == 0 {
385                pgn.push_str(&format!("{}. ", move_number));
386            }
387
388            // Convert move to proper SAN notation
389            let move_san = convert_to_san(&current_board, analysis.chess_move);
390            pgn.push_str(&move_san);
391
392            // Make the move to update the board for the next iteration
393            current_board = current_board.make_move_new(analysis.chess_move);
394
395            // Add evaluation annotation
396            if analysis.centipawn_loss > 50.0 {
397                pgn.push_str("??"); // Blunder
398            } else if analysis.centipawn_loss > 20.0 {
399                pgn.push('?'); // Questionable move
400            }
401
402            // Add evaluation comment
403            pgn.push_str(&format!(
404                " {{ eval: {:.2}, loss: {:.1}cp }} ",
405                analysis.evaluation_after, analysis.centipawn_loss
406            ));
407
408            if i % 2 == 1 {
409                move_number += 1;
410                pgn.push('\n');
411            }
412        }
413
414        pgn.push_str(&format!(" {}\n", result));
415        pgn
416    }

418    fn calculate_statistics(&self) -> (f32, f32, f32, f32) {
419        let chess_vector_moves: Vec<_> = self
420            .move_history
421            .iter()
422            .enumerate()
423            .filter(|(i, _)| {
424                let move_color = if i % 2 == 0 {
425                    Color::White
426                } else {
427                    Color::Black
428                };
429                move_color == self.chess_vector_color
430            })
431            .map(|(_, analysis)| analysis)
432            .collect();
433
434        let stockfish_moves: Vec<_> = self
435            .move_history
436            .iter()
437            .enumerate()
438            .filter(|(i, _)| {
439                let move_color = if i % 2 == 0 {
440                    Color::White
441                } else {
442                    Color::Black
443                };
444                move_color != self.chess_vector_color
445            })
446            .map(|(_, analysis)| analysis)
447            .collect();
448
449        let chess_vector_total_loss: f32 =
450            chess_vector_moves.iter().map(|m| m.centipawn_loss).sum();
451        let chess_vector_good_moves = chess_vector_moves
452            .iter()
453            .filter(|m| m.centipawn_loss < 20.0)
454            .count();
455        let chess_vector_accuracy = if chess_vector_moves.is_empty() {
456            0.0
457        } else {
458            chess_vector_good_moves as f32 / chess_vector_moves.len() as f32 * 100.0
459        };
460
461        let stockfish_total_loss: f32 = stockfish_moves.iter().map(|m| m.centipawn_loss).sum();
462        let stockfish_good_moves = stockfish_moves
463            .iter()
464            .filter(|m| m.centipawn_loss < 20.0)
465            .count();
466        let stockfish_accuracy = if stockfish_moves.is_empty() {
467            0.0
468        } else {
469            stockfish_good_moves as f32 / stockfish_moves.len() as f32 * 100.0
470        };
471
472        (
473            chess_vector_total_loss,
474            chess_vector_accuracy,
475            stockfish_total_loss,
476            stockfish_accuracy,
477        )
478    }

677        }
678    }
679
680    // Calculate centipawn losses
681    calculate_centipawn_losses(&mut game_state.move_history);
682
683    // Game finished
684    println!("\nðŸ Game finished!");
685    match game_state.game.result() {
686        Some(chess::GameResult::WhiteCheckmates) => println!("ðŸ† White wins by checkmate!"),
687        Some(chess::GameResult::BlackCheckmates) => println!("ðŸ† Black wins by checkmate!"),
688        Some(chess::GameResult::WhiteResigns) => println!("ðŸ† Black wins by resignation!"),
689        Some(chess::GameResult::BlackResigns) => println!("ðŸ† White wins by resignation!"),
690        Some(chess::GameResult::Stalemate) => println!("ðŸ¤ Draw by stalemate"),
691        Some(chess::GameResult::DrawAccepted) => println!("ðŸ¤ Draw agreed"),
692        Some(chess::GameResult::DrawDeclared) => println!("ðŸ¤ Draw declared"),
693        None => println!("ðŸ¤ Game incomplete (move limit reached)"),
694    }
695
696    // Show statistics
697    let (cv_loss, cv_accuracy, sf_loss, sf_accuracy) = game_state.calculate_statistics();
698
699    println!("\nðŸ“Š Game Statistics:");
700    println!("Chess Vector Engine:");
701    println!("  - Total centipawn loss: {:.1}", cv_loss);
702    println!("  - Accuracy: {:.1}%", cv_accuracy);
703    println!("\nStockfish:");
704    println!("  - Total centipawn loss: {:.1}", sf_loss);
705    println!("  - Accuracy: {:.1}%", sf_accuracy);
706
707    // Generate and display PGN
708    println!("\nðŸ“‹ PGN:");
709    println!("{}", game_state.generate_pgn());
710
711    // Save PGN to file
712    let filename = format!(
713        "chess_vector_vs_stockfish_safe_{}.pgn",
714        game_state.start_time.format("%Y%m%d_%H%M%S")
715    );
716    std::fs::write(&filename, game_state.generate_pgn())?;
717    println!("ðŸ’¾ Game saved to: {}", filename);
718
719    Ok(())
720}
721
722fn calculate_centipawn_losses(move_history: &mut [MoveAnalysis]) {
723    for i in 0..move_history.len() {
724        if i > 0 {
725            let prev_eval = move_history[i - 1].evaluation_after;
726            let current_eval = move_history[i].evaluation_after;
727
728            // Calculate centipawn loss properly
729            // All evaluations are from White's perspective
730            // So for ANY move, if the evaluation gets worse for the side to move, that's a loss
731            let eval_diff = if i % 2 == 1 {
732                // Odd index = White just moved (0-indexed)
733                // White move: if evaluation decreased, that's bad for white
734                prev_eval - current_eval
735            } else {
736                // Even index = Black just moved
737                // Black move: if evaluation increased (better for white), that's bad for black
738                current_eval - prev_eval
739            };
740
741            // Convert to centipawns and ensure non-negative
742            move_history[i].centipawn_loss = (eval_diff * 100.0).max(0.0);
743        }
744    }
745}
