138/// Generate random positions from actual games for more diverse training
139fn generate_game_positions(num_games: usize) -> Vec<(Board, f32)> {
140    let mut positions = Vec::new();
141    let mut rng_state = 42u64; // Simple PRNG state
142
143    for _game in 0..num_games {
144        let mut game = Game::new();
145        let mut move_count = 0;
146
147        // Play random moves to generate diverse positions
148        while game.result().is_none() && move_count < 30 {
149            let board = game.current_position();
150            let legal_moves: Vec<ChessMove> = MoveGen::new_legal(&board).collect();
151
152            if legal_moves.is_empty() {
153                break;
154            }
155
156            // Simple PRNG (LCG)
157            rng_state = rng_state.wrapping_mul(1103515245).wrapping_add(12345);
158            let move_idx = (rng_state as usize) % legal_moves.len();
159            let chosen_move = legal_moves[move_idx];
160
161            if !game.make_move(chosen_move) {
162                break;
163            }
164
165            move_count += 1;
166
167            // Evaluate position based on material and position
168            let evaluation = evaluate_position_heuristic(&game.current_position());
169            positions.push((game.current_position(), evaluation));
170
171            // Stop if too many positions
172            if positions.len() >= 100 {
173                break;
174            }
175        }
176
177        if positions.len() >= 100 {
178            break;
179        }
180    }
181
182    positions
183}

185/// Simple heuristic evaluation for training data generation
186fn evaluate_position_heuristic(board: &Board) -> f32 {
187    let mut score = 0.0;
188
189    // Material count
190    for square in chess::ALL_SQUARES {
191        if let Some(piece) = board.piece_on(square) {
192            let piece_value = match piece {
193                chess::Piece::Pawn => 1.0,
194                chess::Piece::Knight => 3.0,
195                chess::Piece::Bishop => 3.0,
196                chess::Piece::Rook => 5.0,
197                chess::Piece::Queen => 9.0,
198                chess::Piece::King => 0.0,
199            };
200
201            if board.color_on(square) == Some(Color::White) {
202                score += piece_value;
203            } else {
204                score -= piece_value;
205            }
206        }
207    }
208
209    // Add some positional factors
210    let legal_moves = MoveGen::new_legal(board).count() as f32;
211    if board.side_to_move() == Color::White {
212        score += legal_moves * 0.01;
213    } else {
214        score -= legal_moves * 0.01;
215    }
216
217    // King safety (very basic)
218    if board.checkers().popcnt() > 0 {
219        if board.side_to_move() == Color::White {
220            score -= 1.0;
221        } else {
222            score += 1.0;
223        }
224    }
225
226    // Clamp to reasonable range
227    score.clamp(-10.0, 10.0)
228}

230/// Train NNUE with comprehensive dataset
231fn train_nnue_comprehensive(
232    config: NNUEConfig,
233    epochs: usize,
234    save_interval: usize,
235    output_path: &str,
236    include_game_positions: bool,
237    resume_training: bool,
238) -> Result<(), Box<dyn std::error::Error>> {
239    println!("🧠 Creating NNUE with configuration: {:?}", config);
240    let mut nnue = NNUE::new(config.clone())?;
241
242    // Try to load existing model if resuming training
243    if resume_training {
244        let config_path = format!("{}.config", output_path);
245        if std::path::Path::new(&config_path).exists() {
246            println!("🔄 Resuming training from existing model: {}", output_path);
247            if let Err(e) = nnue.load_model(output_path) {
248                println!("⚠️  Failed to load existing model: {}", e);
249                println!("🆕 Starting fresh training instead");
250            } else {
251                println!("✅ Successfully loaded existing model for incremental training");
252            }
253        } else {
254            println!(
255                "📝 No existing model found at {}, starting fresh training",
256                output_path
257            );
258        }
259    } else {
260        println!("🆕 Starting fresh training (not resuming from existing model)");
261    }
262
263    // Generate training data
264    println!("📊 Generating training positions...");
265    let mut training_data = generate_training_positions();
266    println!("   Generated {} curated positions", training_data.len());
267
268    if include_game_positions {
269        println!("🎮 Generating positions from random games...");
270        let game_positions = generate_game_positions(50);
271        println!("   Generated {} game positions", game_positions.len());
272        training_data.extend(game_positions);
273    }
274
275    println!(
276        "📈 Total training dataset: {} positions",
277        training_data.len()
278    );
279
280    // Training loop
281    println!("🚀 Starting NNUE training for {} epochs...", epochs);
282    let start_time = Instant::now();
283
284    for epoch in 0..epochs {
285        let epoch_start = Instant::now();
286
287        // Use incremental training if resuming, otherwise regular training
288        let loss = if resume_training {
289            nnue.incremental_train(&training_data, true)?
290        } else {
291            nnue.train_batch(&training_data)?
292        };
293
294        let epoch_time = epoch_start.elapsed();
295        println!(
296            "Epoch {}/{}: Loss = {:.6}, Time = {:.2}s",
297            epoch + 1,
298            epochs,
299            loss,
300            epoch_time.as_secs_f32()
301        );
302
303        // Save intermediate models
304        if save_interval > 0 && (epoch + 1) % save_interval == 0 {
305            let checkpoint_path = format!("{}_epoch_{}", output_path, epoch + 1);
306            nnue.save_model(&checkpoint_path)?;
307            println!("💾 Saved checkpoint: {}", checkpoint_path);
308        }
309
310        // Early stopping if loss is very low
311        if loss < 0.001 {
312            println!("✅ Early stopping: Loss target reached");
313            break;
314        }
315    }
316
317    let total_time = start_time.elapsed();
318    println!(
319        "🏁 Training completed in {:.2}s ({:.2}s per epoch average)",
320        total_time.as_secs_f32(),
321        total_time.as_secs_f32() / epochs as f32
322    );
323
324    // Save final model
325    nnue.save_model(output_path)?;
326    println!("💾 Final model saved: {}", output_path);
327
328    // Test the trained model
329    println!("🧪 Testing trained model...");
330    test_nnue_model(&mut nnue)?;
331
332    Ok(())
333}

369/// Load and test an existing NNUE model
370fn test_existing_model(model_path: &str) -> Result<(), Box<dyn std::error::Error>> {
371    println!("📂 Loading existing NNUE model: {}", model_path);
372
373    let config = NNUEConfig::default();
374    let mut nnue = NNUE::new(config)?;
375
376    if Path::new(&format!("{}.config", model_path)).exists() {
377        nnue.load_model(model_path)?;
378        println!("✅ Model loaded successfully");
379
380        test_nnue_model(&mut nnue)?;
381    } else {
382        return Err(format!("Model file not found: {}.config", model_path).into());
383    }
384
385    Ok(())
